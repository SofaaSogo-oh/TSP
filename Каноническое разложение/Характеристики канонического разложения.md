# Связано с
[[Каноническое разложение СП]]
# Математическое ожидание
$$
M[x(t)] = m_x(t) + \sum M[v_k]\varphi_k(t) = m_x(t)
$$
# Корреляционная формула
$$
R_x(t_1,t_2) = \sum_{k=1}^{n}D(v_k)\varphi_k(t_1)\varphi_k(t_2) 
$$
![[Корреляционная формула канонического разложения]]
# Дисперсия
$$
D_x(t)=\left.R_x(t_1,t_2)\right|_{t=t_1=t_2}=\sum_{k=1}^n D(v_k)\varphi_k^2(t)
$$
# Производная 
$$
x'(t) = m'(t) + \sum_{k=1}^{n}v_k\varphi'_k(t)
$$
# Интеграл
$$
\int_{t_0}^t x(t)\,dt=\int_{t_0}^t m_x(t)\,dt + \sum_{k=1}^n v_k \int_{t_0}^t \varphi(t)\,dt
$$
